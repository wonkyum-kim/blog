## 카드 정리 1

조커 박스는 다음과 같이 설정할 수 있다.

- 어떤 박스가 카드를 두 종류 이상 가지고 있다면, 조커 박스로 만든다.
- 모든 박스가 카드를 한 종류 가지고 있으면, 임의의 박스를 조커로 만든다.

function check(v, m) {
  const temp = new Array(m).fill(0)
  for (let i = 0; i < m; ++i) {
    for (let j = 0; j < v.length; ++j) {
      temp[i] += v[j][i] > 0 ? 1 : 0
    }
  }

  let sum = 0
  for (let i = 0; i < m; ++i) {
    if (temp[i] > 1) sum += temp[i] - 1
  }

  return sum
}

function solve() {
  const [n, m] = [0, 0].map(() => input.getNumber)

  let ans = 0
  const v = []

  for (let i = 0; i < n; ++i) {
    const temp = []
    let count = 0
    for (let j = 0; j < m; ++j) {
      const num = input.getNumber
      if (num !== 0) count++
      temp.push(num)
    }
    // 카드가 한 종류만 들어있는 경우
    if (count === 1) {
      v.push(temp)
    }
    // 카드가 두 종류 이상 있는 경우
    else if (count > 1) {
      ans++
    }
  }

  // 두 종류 이상 가지고 있는 박스 중 하나를 조커로 만든다.
  if (ans > 0) ans--
  // 모든 박스가 한 종류의 카드만 가지고 있는 경우 임의의 박스를 조커로 만든다.
  else v.pop()

  // 한 종류 가지고 있는 박스 중에서 조커로 이동 시킬수 있으면 이동시킨다.
  ans += check(v, m)

  console.log(ans)
}

solve()
