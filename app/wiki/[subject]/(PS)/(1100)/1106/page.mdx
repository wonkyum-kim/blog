## 호텔

[아이템의 개수가 무한 개인 배낭 문제](/wiki/Algorithm/Knapsack-problem)다.

따라서 앞에서부터 dp 배열을 갱신해도 문제가 발생하지 않는다.

<CodeSelector names={['1106.js']} desc='ex1' />

```js showLineNumbers
function solve() {
  const c = input.getNumber
  const n = input.getNumber
  // dp[i] = i명 일때 최소 비용
  const dp = new Array(1101).fill(Infinity)
  dp[0] = 0

  for (let i = 0; i < n; ++i) {
    const [weight, value] = [0, 0].map(() => input.getNumber)
    for (let j = value; j <= 1100; ++j) {
      dp[j] = Math.min(dp[j], dp[j - value] + weight)
    }
  }

  let ans = Infinity
  for (let i = c; i <= 1100; ++i) {
    ans = Math.min(ans, dp[i])
  }
  console.log(ans)
}

solve()
```
